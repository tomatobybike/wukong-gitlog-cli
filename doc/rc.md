import yaml from 'js-yaml'

const RC_FILES = [
  '.wukongrc',
  '.wukongrc.json',
  '.wukongrc.yaml',
  '.wukongrc.yml',
  '.wukongrc.js'
]

function loadOne(file) {
  if (!fs.existsSync(file)) return null
  if (file.endsWith('.js')) return require(file)
  const content = fs.readFileSync(file, 'utf-8')
  return file.endsWith('.yaml') || file.endsWith('.yml')
    ? yaml.load(content)
    : JSON.parse(content)
}

export function loadWukongConfig(cwd = process.cwd()) {
  const home = process.env.HOME
  let config = {}

  for (const base of [home, cwd]) {
    if (!base) continue
    for (const name of RC_FILES) {
      const file = path.join(base, name)
      const data = loadOne(file)
      if (data) config = { ...config, ...data }
    }
  }

  return config
}

import fs from 'fs'
import path from 'path'
import yaml from 'js-yaml'

const RC_FILES = [
  '.wukongrc',
  '.wukongrc.json',
  '.wukongrc.yaml',
  '.wukongrc.yml',
  '.wukongrc.js'
]

function readFile(file) {
  if (!fs.existsSync(file)) return null
  if (file.endsWith('.js')) return require(file)
  const raw = fs.readFileSync(file, 'utf-8')
  if (file.endsWith('.yaml') || file.endsWith('.yml')) {
    return yaml.load(raw)
  }
  return JSON.parse(raw)
}

export function loadWukongConfig(cwd = process.cwd()) {
  const home = process.env.HOME
  let config = {}

  for (const base of [home, cwd]) {
    if (!base) continue
    for (const name of RC_FILES) {
      const file = path.join(base, name)
      const data = readFile(file)
      if (data) config = deepMerge(config, data)
    }
  }

  return config
}

function deepMerge(a, b) {
  const out = { ...a }
  for (const k in b) {
    out[k] =
      typeof b[k] === 'object' && !Array.isArray(b[k])
        ? deepMerge(a[k] || {}, b[k])
        : b[k]
  }
  return out
}

export function buildWeeklyReport({
  commits,
  overtime,
  opts
}) {
  const total = commits.length
  const authors = {}

  for (const c of commits) {
    const key = c.authorName || 'unknown'
    authors[key] = (authors[key] || 0) + 1
  }

  const topAuthors = Object.entries(authors)
    .sort((a, b) => b[1] - a[1])
    .slice(0, opts.topAuthors || 5)

  return {
    period: 'Weekly',
    totalCommits: total,
    topAuthors,
    overtime
  }
}
